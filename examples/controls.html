Player Controls

Configure play buttons, time display, volume slider, and playback rate selector. Three button styles: circle, shield, minimal.

from wavesurf import Controls controls = Controls( show_play_button=True, show_time=True, show_volume=True, show_playback_rate=True, play_button_style="shield", ) WaveSurfer(audio=audio, sr=24000, controls=controls)
