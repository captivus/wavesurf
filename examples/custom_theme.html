Custom Theme

Build a fully customized theme with gradient waveforms, shield-style play button, title markers, and accent lines.

from wavesurf import Theme, WaveSurfer custom = Theme( wave_color=["#e63946", "#d62839"], progress_color=["#457b9d", "#1d3557"], background="#1d3557", play_button_style="shield", title_marker_color="#e63946", top_accent="linear-gradient(90deg, #e63946, #457b9d)", ) WaveSurfer(audio=audio, sr=24000, title="Branded", theme=custom)
